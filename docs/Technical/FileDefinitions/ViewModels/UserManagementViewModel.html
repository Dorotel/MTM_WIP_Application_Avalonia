LoadUsersCommand = ReactiveCommand.CreateFromTask(async () =>
{
    try
    {
        IsLoading = true;
        
        // Check permissions first
        if (!CanManageUsers)
        {
            throw new UnauthorizedAccessException("Insufficient permissions for user management");
        }
        
        var users = await _userService.GetAllUsersAsync();
        
        Users.Clear();
        foreach (var user in users)
        {
            Users.Add(new UserViewModel(user));
        }
        
        await LoadRolesAndDepartmentsAsync();
        
        Logger.LogInformation("Loaded {Count} users", Users.Count);
    }
    catch (Exception ex)
    {
        Logger.LogError(ex, "Failed to load users");
        throw;
    }
    finally
    {
        IsLoading = false;
    }
});

üîê Security & Authorization

Permission Validation

// Permission Checking Service
private bool HasPermission(string permission)
{
    var currentUser = _applicationState.CurrentUser;
    return currentUser?.Permissions.Contains(permission) == true;
}

// Command Security Decorators
AddUserCommand = ReactiveCommand.CreateFromTask(async () =>
{
    if (!HasPermission("USER_MANAGEMENT"))
    {
        throw new UnauthorizedAccessException("Insufficient permissions to add users");
    }
    
    await ShowAddUserDialog();
}, this.WhenAnyValue(vm => vm.CanManageUsers));

// Hierarchical Permission Checking
private bool CanEditUser(UserViewModel targetUser)
{
    var currentUser = _applicationState.CurrentUser;
    
    // Cannot edit users with higher or equal privileges
    if (targetUser.Role >= currentUser.Role)
    {
        return false;
    }
    
    // Cannot edit own account through this interface
    if (targetUser.UserId == currentUser.UserId)
    {
        return false;
    }
    
    return HasPermission("USER_MANAGEMENT");
}

// Audit Logging for Security Actions
private async Task LogSecurityAction(string action, string targetUserId, string details = null)
{
    var auditEntry = new SecurityAuditEntry
    {
        Action = action,
        PerformedBy = _applicationState.CurrentUser.UserId,
        TargetUserId = targetUserId,
        Timestamp = DateTime.UtcNow,
        Details = details,
        IPAddress = _applicationState.UserIPAddress
    };
    
    await _userService.LogSecurityActionAsync(auditEntry);
}

Security Features

  • Role-Based Access Control: Hierarchical permission system
  • Privilege Escalation Prevention: Cannot edit higher-privilege users
  • Self-Management Protection: Cannot modify own account
  • Action Auditing: All administrative actions logged
  • Session Validation: Continuous verification of user permissions
  • Data Masking: Sensitive data hidden in UI and exports

// User Creation Dialog
private async Task ShowAddUserDialog()
{
    var dialogViewModel = new UserEditDialogViewModel(_userService)
    {
        Title = "Add New User",
        IsNewUser = true,
        AvailableRoles = AvailableRoles.Where(r => r.Level <= CurrentUserRole.Level).ToList(),
        AvailableDepartments = AvailableDepartments
    };
    
    var dialog = new UserEditDialog { DataContext = dialogViewModel };
    var result = await dialog.ShowDialog<bool?>(GetParentWindow());
    
    if (result == true)
    {
        try
        {
            var newUser = await _userService.CreateUserAsync(dialogViewModel.ToUserModel());
            Users.Add(new UserViewModel(newUser));
            
            await LogSecurityAction("USER_CREATED", newUser.UserId, $"Role: {newUser.Role}");
            ShowSuccessMessage($"User {newUser.FullName} created successfully");
        }
        catch (Exception ex)
        {
            Logger.LogError(ex, "Failed to create user");
            ShowErrorMessage($"Failed to create user: {ex.Message}");
        }
    }
}

// Password Reset Dialog
private async Task ShowPasswordResetDialog(UserViewModel user)
{
    var confirmation = await ShowConfirmationDialog(
        "Reset Password",
        $"Are you sure you want to reset the password for {user.FullName}? " +
        "They will be required to change it on next login.");
    
    if (confirmation)
    {
        try
        {
            var tempPassword = await _userService.ResetPasswordAsync(user.UserId);
            
            var resultDialog = new PasswordResetResultDialog
            {
                UserName = user.FullName,
                TemporaryPassword = tempPassword
            };
            
            await resultDialog.ShowDialog(GetParentWindow());
            
            await LogSecurityAction("PASSWORD_RESET", user.UserId);
            ShowSuccessMessage($"Password reset for {user.FullName}");
        }
        catch (Exception ex)
        {
            Logger.LogError(ex, "Failed to reset password for user {UserId}", user.UserId);
            ShowErrorMessage($"Failed to reset password: {ex.Message}");
        }
    }
}

üö® Error Handling

Security-Aware Error Handling

// Security Exception Handling
LoadUsersCommand.ThrownExceptions
    .Subscribe(ex =>
    {
        var userMessage = ex switch
        {
            UnauthorizedAccessException => 
                "You don't have permission to access user management.",
            SecurityException secEx => 
                $"Security error: {secEx.Message}",
            DatabaseConnectionException => 
                "Cannot connect to user database. Please try again.",
            ValidationException valEx => 
                $"Validation error: {valEx.Message}",
            _ => "Failed to load users. Please contact support."
        };
        
        Logger.LogError(ex, "User management operation failed");
        ShowErrorMessage(userMessage);
        
        // Log security-related errors for audit
        if (ex is UnauthorizedAccessException or SecurityException)
        {
            _ = LogSecurityAction("UNAUTHORIZED_ACCESS_ATTEMPT", null, ex.Message);
        }
    });

// Safe Error Messages (no sensitive info)
private string GetSafeErrorMessage(Exception ex)
{
    // Never expose database schemas, internal errors, or sensitive data
    return ex switch
    {
        ArgumentException => "Invalid input provided",
        DuplicateKeyException => "User already exists with this employee ID",
        ReferentialIntegrityException => "Cannot delete user with existing transactions",
        _ => "Operation failed. Please contact support."
    };
}
