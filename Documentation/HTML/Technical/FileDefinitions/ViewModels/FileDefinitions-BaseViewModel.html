protected virtual void Dispose(bool disposing)
{
    if (disposing && !_disposed)
    {
        Logger.LogDebug("Disposing BaseViewModel: {Type}", GetType().Name);
        
        // Dispose all managed reactive subscriptions
        _disposables?.Dispose();
        
        _disposed = true;
    }
}

üîß Usage Patterns

üìù Typical Derived Class Implementation

public class AddItemViewModel : BaseViewModel
{
    public AddItemViewModel(
        IInventoryService inventoryService,
        ILogger<AddItemViewModel> logger) : base(logger)
    {
        // Null guard validations
        _inventoryService = inventoryService ?? throw new ArgumentNullException(nameof(inventoryService));
        
        // Initialize commands and reactive subscriptions
        InitializeCommands();
        SetupReactiveSubscriptions();
    }
    
    private void SetupReactiveSubscriptions()
    {
        // Add subscriptions to Disposables for automatic cleanup
        this.WhenAnyValue(x => x.SomeProperty)
            .Subscribe(value => HandlePropertyChange(value))
            .DisposeWith(Disposables);
    }
}

üìã MTM Best Practices

üö® Critical Implementation Rules

  • Always call base constructor: Ensure logging is properly initialized
  • Use Disposables collection: Add all reactive subscriptions for proper cleanup
  • Null guard all dependencies: Validate constructor parameters
  • Log significant operations: Use Logger for debugging and audit trails

‚ö†Ô∏è Common Pitfalls to Avoid

  • Memory Leaks: Always dispose reactive subscriptions
  • Missing Logging: Don't skip logger parameter validation
  • Incorrect Disposal: Don't override Dispose() without calling base
  • Threading Issues: Use UI thread for property changes
